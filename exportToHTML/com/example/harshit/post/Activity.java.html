28   import android.content.Context; 
29   import android.content.Intent; 
30   import android.net.ConnectivityManager; 
31   import android.net.NetworkInfo; 
32   import android.support.v7.app.AppCompatActivity; 
33   import android.os.Bundle; 
34   import android.view.View; 
35   import android.widget.Button; 
36   import android.widget.TextView; 
37    
38   public class Activity extends AppCompatActivity { 
39    
40       Button showData, addData,selectSeat,layoutGenerator,displayClassroom; // declaration of display buttons on the main screen . 
41       TextView networkStatus;   // declaration of text-view of network status 
42    
43    
44       @Override 
45       protected void onCreate(Bundle savedInstanceState) { 
46    
47           super.onCreate(savedInstanceState);   // Calling the constructor of parent class. 
48           setContentView(R.layout.activity_);   // Setting the activity content to the main view. 
49    
50           // **** 
51           // assigning the display button and text resource to the declared variable 
52           showData  = findViewById(R.id.showData); 
53           addData = findViewById(R.id.addData); 
54           selectSeat = findViewById(R.id.selectSeat); 
55           layoutGenerator = findViewById(R.id.layoutGenerator); 
56           displayClassroom = findViewById(R.id.displayClassroom); 
57           networkStatus = findViewById(R.id.Network_status); 
58           // **** 
59    
60    
61           // checking the internet connectivity of the device 
62           ConnectivityManager connectivityManager = (ConnectivityManager)getSystemService(Context.CONNECTIVITY_SERVICE); 
63           NetworkInfo networkInfo = connectivityManager.getActiveNetworkInfo(); // fetching network information 
64    
65           if(networkInfo!=null &&networkInfo.isConnected()) 
66           { 
67               networkStatus.setVisibility(View.INVISIBLE); 
68           } 
69           else 
70           { 
71               // disabled buttons if internet is not connected 
72               showData.setEnabled(false); 
73               addData.setEnabled(false); 
74               selectSeat.setEnabled(false); 
75               displayClassroom.setEnabled(false); 
76               layoutGenerator.setEnabled(false); 
77    
78           } 
79       } 
80    
81       //**** 
82       // Declaring the on-click screens(activity) of the display buttons 
83       public void addData(View view) 
84       { 
85           startActivity(new Intent(this,addData.class)); 
86       } 
87       public void showData(View view) 
88       { 
89           startActivity(new Intent(this,showData.class)); 
90       } 
91       public void selectSeat(View view) {startActivity(new Intent(this,selectSeat.class));} 
92       public void layoutGenerator(View view) {startActivity(new Intent(this,Layout.class));} 
93       public void displayClassroom(View view) {startActivity(new Intent(this,display.class));} 
94       //**** 
95   }
