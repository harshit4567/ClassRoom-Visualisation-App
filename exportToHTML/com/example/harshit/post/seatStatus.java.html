113      private void getJSON(final String urlWebService) { 
114         /* 
115         * As fetching the json string is a network operation 
116         * And we cannot perform a network operation in main thread 
117         * so we need an AsyncTask 
118         * The constrains defined here are 
119         * Void -> We are not passing anything 
120         * Void -> Nothing at progress update as well 
121         * String -> After completion it should return a string and it will be the json string 
122         * */ 
123          class GetJSON extends AsyncTask<Void, Void, String> { 
124   
125              //this method will be called before execution 
126              //you can display a progress bar or something 
127              //so that user can understand that he should wait 
128              //as network operation may take some time 
129              @Override 
130              protected void onPreExecute() { 
131                  super.onPreExecute(); 
132              } 
133   
134              //this method will be called after execution 
135              //so here we are displaying a toast with the json string 
136              @Override 
137              protected void onPostExecute(String s) { 
138                  super.onPostExecute(s); 
139   
140                  try { 
141                      loadIntoListView(s); 
142                  } catch (JSONException e) { 
143                      e.printStackTrace(); 
144                  } 
145   
146              } 
147   
148              //in this method we are fetching the json string 
149              @Override 
150              protected String doInBackground(Void... voids) { 
151   
152   
153                  try { 
154                      //creating a URL 
155                      URL url = new URL(urlWebService); 
156   
157                      //Opening the URL using HttpURLConnection 
158                      HttpURLConnection con = (HttpURLConnection) url.openConnection(); 
159   
160                      //StringBuilder object to read the string from the service 
161                      StringBuilder sb = new StringBuilder(); 
162   
163                      //We will use a buffered reader to read the string from service 
164                      BufferedReader bufferedReader = new BufferedReader(new InputStreamReader(con.getInputStream())); 
165   
166                      //A simple string to read values from each line 
167                      String json; 
168   
169                      //reading until we don't find null 
170                      while ((json = bufferedReader.readLine()) != null) { 
171   
172                          //appending it to string builder 
173                          sb.append(json).append("\n"); 
174                      } 
175   
176                      //finally returning the read string 
177                      return sb.toString().trim(); 
178                  } catch (Exception e) { 
179                      return null; 
180                  } 
181   
182              } 
183          } 
184   
185          //creating asynctask object and executing it 
186          GetJSON getJSON = new GetJSON(); 
187          getJSON.execute(); 
188      }

190      private void loadIntoListView(String json) throws JSONException { 
191          //creating a json array from the json string 
192          JSONArray jsonArray = new JSONArray(json); 
193   
194          //creating a string array for listview 
195          String[] seatStatus = new String[jsonArray.length()]; 
196   
197          //looping through all the elements in json array 
198          for (int i = 0; i < jsonArray.length(); i++) { 
199   
200              //getting json object from the json array 
201              JSONObject obj = jsonArray.getJSONObject(i); 
202              //getting the name from the json object and putting it inside string array 
203              seatStatus[i] = obj.getString("seat_number") ; 
204   
205          } 
206   
207          //the array adapter to load data into list 
208          ArrayAdapter<String> arrayAdapter = new ArrayAdapter<String>(this, android.R.layout.simple_list_item_1, seatStatus); 
209   
210          //attaching adapter to list view 
211          GridView.setAdapter(arrayAdapter); 
212          GridView.setOnItemClickListener(new AdapterView.OnItemClickListener() {

255      class BackgroundTask extends AsyncTask<String, Void, String> { 
256   
257          String SELECT_SEAT_URL; 
258          @Override 
259          protected void onPreExecute(){ 
260   
261              SELECT_SEAT_URL = "http://22harshit.000webhostapp.com/new_seat.php"; 
262          } 
263   
264   
265          @Override 
266          protected String doInBackground(String... args) { 
267              String id,seat_number; 
268              Random rnd = new Random(); 
269              String state = Integer.toString(rnd.nextInt(10) + 1); 
270              id = args[0]; 
271              seat_number = args[1]; 
272              try { 
273                  //creating a URL 
274                  URL url  = new URL(SELECT_SEAT_URL); 
275                  //Opening the URL using HttpURLConnection 
276                  HttpURLConnection httpURLConnection  = (HttpURLConnection) url.openConnection(); 
277                  // specifying the request method 
278                  httpURLConnection.setRequestMethod("POST"); 
279                  httpURLConnection.setDoOutput(true); 
280                  // creating a output stream 
281                  OutputStream outputStream = httpURLConnection.getOutputStream(); 
282                  //We will use a buffered reader to read the string from service 
283                  BufferedWriter bufferedWriter = new BufferedWriter(new OutputStreamWriter(outputStream,"UTF-8")); 
284                  // encoding the data string to be sent 
285                  String data_string = URLEncoder.encode("ID","UTF-8")+"="+URLEncoder.encode(id,"UTF-8")+"&"+ 
286                          URLEncoder.encode("seat_number","UTF-8")+"="+URLEncoder.encode(seat_number,"UTF-8")+"&"+ 
287                          URLEncoder.encode("state","UTF-8")+"="+URLEncoder.encode(state,"UTF-8"); 
288                  // writing the data to the buffer 
289                  bufferedWriter.write(data_string); 
290   
291                  // closing and flushing the connections 
292                  bufferedWriter.flush(); 
293                  bufferedWriter.close(); 
294                  bufferedWriter.close(); 
295                  outputStream.close(); 
296   
297                  // reading the response from server 
298                  InputStream inputStream = httpURLConnection.getInputStream(); 
299                  BufferedReader r = new BufferedReader(new InputStreamReader(inputStream)); 
300                  StringBuilder statusMessage = new StringBuilder(); 
301                  String line; 
302                  while ((line = r.readLine()) != null) { 
303                      statusMessage.append(line).append('\n'); 
304                  } 
305                  inputStream.close(); 
306                  httpURLConnection.disconnect(); 
307   
308                  //finally returning the status string 
309                  return statusMessage.toString(); 
310   
311              } catch (MalformedURLException e) { 
312                  e.printStackTrace(); 
313              } catch (IOException e) { 
314                  e.printStackTrace(); 
315              } 
316              return null; 
317          }
